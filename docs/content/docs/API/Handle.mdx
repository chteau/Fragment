import { TypeTable } from 'fumadocs-ui/components/type-table';
import { Callout } from 'fumadocs-ui/components/callout';

<Callout title="Client Only" type="info">
    This class is only available when running on the client.
</Callout>

A Handle is the core primitive of Fragment. It binds a renderer function to a specific UI Instance resolved via a path from `PlayerGui`, and re-runs that renderer whenever its subscribed data or state changes.

Handles manage:

- Rendering and effect lifecycles
- Local reactive state slots
- Derived and memoized values
- Async state helpers
- Named signal connections
- Store and Context subscriptions
- Diffing containers for dynamic children

---

# Render-cycle methods

These methods are only active **inside a renderer**. Calling them outside produces a warning and does nothing.

## `Effect(fn)`

Registers a side-effect that runs after every render. If `fn` returns a cleanup function, it is called before the next render.

<TypeTable
  type={{
    Effect: {
      description: "Registers a side-effect. The returned cleanup function runs before the next render.",
      type: "(fn: () -> (() -> ())?) -> ()"
    }
  }}
/>

**Example**:
```luau title="ClockHandle.luau"
Handle.Effect(function()
    local conn = RunService.Heartbeat:Connect(function()
        element.Text = os.date("%H:%M:%S")
    end)

    return function()
        conn:Disconnect() -- [!code highlight]
    end
end)
```

---

## `Action(actionName, fn)`

Registers a named action callable from outside the renderer via the handle.

<TypeTable
  type={{
    Action: {
      description: "Registers a named action callable from outside the renderer.",
      type: "(actionName: string, fn: (...any) -> any) -> ()"
    }
  }}
/>

**Example**:
```luau title="ShopHandle.luau"
Handle.Action("Toggle", function(state: boolean) -- [!code highlight]
    element.Visible = state
end)

-- From anywhere:
Fragment.getHandle("Shop"):Toggle(true)
```

## `Once(fn)`

Runs `fn` exactly once on the first render.

<TypeTable
  type={{
    Once: {
      description: "Runs a function exactly once per Handle lifetime.",
      type: "(fn: () -> ()) -> ()"
    }
  }}
/>

**Example**:
```luau title="IndexWindow.luau"
Handle.Once(function()
    for index, mutation in ipairs(MutationsAndRarity.Mutations) do -- [!code highlight]
        table.insert(tabs, { Name = mutation.Name, Tier = index }) -- [!code highlight]
    end -- [!code highlight]
end)
```

## `Connect(connName, signal, callback)`

Binds a named connection slot to an `RBXScriptSignal`.

<TypeTable
  type={{
    Connect: {
      description: "Connects to an RBXScriptSignal under a named slot. Replaces any previous connection with the same name.",
      type: "(connName: string, signal: RBXScriptSignal, callback: (...any) -> ()) -> ()"
    }
  }}
/>

**Example**:
```luau title="ShopHandle.luau"
local closeBtn = element.TopFrame.Close :: TextButton

Handle.Connect("Close", closeBtn.Activated, function() -- [!code highlight]
    Fragment.getHandle("Shop"):Toggle(false)
end)
```

## `RegisterContainer(frame)`

Returns a persistent [Container](/docs/API/Container) for `frame` that diffs children across renders.

<TypeTable
  type={{
    RegisterContainer: {
      description: "Returns a persistent diffing Container for the given frame.",
      type: "(frame: Instance) -> Container"
    }
  }}
/>

**Example**:
```luau title="LeaderboardHandle.luau"
local list = Handle.RegisterContainer(element.ScrollFrame) -- [!code highlight]

for i, entry in ipairs(data.entries) do
    local live = list.UpsertChild(entry.id, Assets.Row:Clone())
    live.Label.Text  = entry.name
    live.LayoutOrder = i
end
```

# Bindings

Bindings are slots that persist across renders. They must be called **in the same order on every render**.

## `handle:bind(initialValue)`

Creates a reactive state slot inside the renderer. The slot is initialised once on the first render.

<TypeTable
  type={{
    bind: {
      description: "Creates a reactive state slot. Returns a getter and a setter. The setter triggers a deferred re-render if the value changed.",
      type: "(initialValue: any) -> (() -> any, (any) -> ())"
    }
  }}
/>

Returns:
- **getter**: `() -> any` — reads the current value
- **setter**: `(any) -> ()` — updates the value and queues a re-render

**Example**:
```luau title="TabsHandle.luau"
local tabIndex, setTabIndex = Handle:bind(1) -- [!code highlight]

for i, tab in ipairs(tabs) do
    local live = container.UpsertChild(tab.name, Assets.Tab:Clone())
    live.Active = tabIndex() == i

    Handle.Connect("Tab_" .. tab.name, live.Activated, function()
        setTabIndex(i)
    end)
end
```

## `handle:derive(fn, deps)`

Returns a getter whose value is recomputed only when `deps` changes. Use this to avoid recomputing expensive values on every render.

<TypeTable
  type={{
    derive: {
      description: "Returns a getter that recomputes its value only when deps changes.",
      type: "(fn: () -> any, deps: { any }) -> () -> any"
    }
  }}
/>

**Example**:
```luau title="IndexWindow.luau"
local getFiltered = Handle:derive(function() -- [!code highlight]
    return superDuperTableFilterFunction(allItems, function(item) -- [!code highlight]
        return item.tier == currentTier -- [!code highlight]
    end) -- [!code highlight]
end, { currentTier }) -- [!code highlight]

for _, item in ipairs(getFiltered()) do
    -- render item
end
```

---

## `handle:hold(fn, deps)`

Returns a stable function reference. The same function object is returned as long as `deps` is unchanged.

<TypeTable
  type={{
    hold: {
      description: "Returns a stable function reference. Only replaced when deps changes.",
      type: "(fn: (...any) -> any, deps: { any }) -> (...any) -> any"
    }
  }}
/>

**Example**:
```luau title="ShopHandle.luau"
local onBuy = Handle:hold(function() -- [!code highlight]
    purchaseItem(selectedItem()) -- [!code highlight]
end, { selectedItem() }) -- [!code highlight]

Handle.Connect("BuyBtn", buyButton.Activated, onBuy)
```

## `handle:await(fn, deps?)`

Runs `fn` asynchronously and returns a status table. Triggers a re-render when the operation settles. Re-runs whenever `deps` changes.

<Callout title="Non-blocking" type="info">
    `await` uses `task.spawn` internally and never yields the renderer. The renderer immediately receives a `"pending"` status table and re-renders again once the operation resolves.
</Callout>

<TypeTable
  type={{
    await: {
      description: "Runs fn asynchronously and returns a reactive status table.",
      type: "(fn: () -> any, deps: { any }?) -> { status: string, value: any, error: any }"
    }
  }}
/>

`status` is one of:
- `"pending"` — operation in progress
- `"resolved"` — completed successfully, `value` holds the result
- `"rejected"` — failed, `error` holds the message

**Example**:
```luau title="ShopHandle.luau"
local inventory = Handle:await(function() -- [!code highlight]
    return InventoryService:FetchAsync() -- [!code highlight]
end, { store.lastUpdated }) -- [!code highlight]

if inventory.status == "pending" then
    element.Spinner.Visible = true
    return
end

element.Spinner.Visible = false

if inventory.status == "resolved" then
    for _, item in ipairs(inventory.value) do
        -- render item
    end
end
```

## `handle:pull(ctx)`

Subscribes the handle to a [Context](/docs/API/Context) and returns the live value table. The handle re-renders whenever `ctx:patch()` is called.

<TypeTable
  type={{
    pull: {
      description: "Subscribes to a Context and returns its live value table.",
      type: "(ctx: Context) -> { [string]: any }"
    }
  }}
/>

**Example**:
```luau title="ShopHandle.luau"
local ctx = Handle:pull(WindowContext) -- [!code highlight]

element.Visible = win.active == "Shop"
```

# Module-level state

## `Handle.useState(initialValue)`

Allocates a permanent state slot at **module level**, outside the renderer. Returns a getter and setter that remain stable for the entire lifetime of the handle.

<Callout title="Module level only" type="warn">
    Call `useState` once at the top of your module, not inside the renderer. Use `handle:bind()` instead when the state only makes sense inside the renderer.
</Callout>

<TypeTable
  type={{
    useState: {
      description: "Allocates a permanent state slot at module level.",
      type: "(initialValue: any) -> (() -> any, (any) -> ())"
    }
  }}
/>

**Example**:
```luau title="ShopHandle.luau"
local Handle = Fragment.newHandle("Shop", { "PlayerGui", "Shop" })

local tabIndex, setTabIndex = Handle.useState(1) -- [!code highlight]

return Handle(function(element: Frame)
    local current = tabs[tabIndex()]
    -- ...
end)
```
