- Declarative UI Handles
- Reactive local bindings
- Derived state
- Async state helpers
- Global stores
- Context propagation
- Automatic effect lifecycle management

# Properties

## `Fragment.DELETE`

Sentinel value used to delete context keys.

<TypeTable
  type={{
    DELETE: {
      description: "Sentinel value used to delete context keys.",
      type: "any"
    }
  }}
/>


**Example**:
```luau title="WindowContext.luau"
WindowContext._value.toggle = function(name: string)
    if WindowContext._value.active == name then
        WindowContext:patch({ active = Fragment.DELETE }) -- Sets active to "nil" [!code highlight]
    else
        WindowContext:patch({ active = name })
    end
end
```

# Methods

## `createStore(name, builder)`

Creates a new reactive store.

<TypeTable
  type={{
    createStore: {
      description: "Creates a new reactive store.",
      type: "(name: string, builder: (set: (partial: any) -> (), get: () -> any) -> any) -> any"
    }
  }}
/>

**Example**:
```luau title="ReplicaServiceStore.luau"
return Fragment.createStore("ReplicaServiceStore", function(set, get)
    return {
        ReplicaData = {},

        setData = function(newData: table)
            set({
                ReplicaData = newData,
            })
        end,

        getData = function()
            return get().ReplicaData
        end,

        -- ...
    }
end)
```
